String getName(int level) #

static String getName(int level) {
  switch(level) {
    case L:
      return 'Low';
    case M:
      return 'Medium';
    case Q:
      return 'Quality';
    case H:
      return 'High';
    default:
      throw 'not supported';
  }
}
