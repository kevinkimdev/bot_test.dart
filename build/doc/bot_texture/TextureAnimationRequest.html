void update() #

void update() {
  if(_frame == null) {
    _frame = -_delay;
  } else if(_frame < (_frameCount - 1)){
    _frame++;
    assert(_frame < _frameCount);
  } else {
    _done = true;
  }

  if(_frame == 0 && !_done) {
    _startEventHandle.fireEvent(EventArgs.empty);
  }
}
