Code void fail(String messageText, String stack) #

void fail(String messageText, String stack) {
  if (result != null) {
    if (result == PASS) {
      error('Test failed after initially passing: $messageText', stack);
    } else if (result == FAIL) {
      error('Test failed more than once: $messageText', stack);
    }
  } else {
    result = FAIL;
    message = messageText;
    stackTrace = stack;
    _complete();
  }
}

Code void run() #

void run() {
  if (enabled) {
    result = stackTrace = null;
    message = '';
    _doneTeardown = false;
    if (_setUp != null) {
      _setUp();
    }
    _config.onTestStart(this);
    startTime = new Date.now();
    runningTime = null;
    test();
  }
}
