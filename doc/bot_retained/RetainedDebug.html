Code void borderElements(stage) #

static void borderElements(stage) {
  var ctx = stage.ctx;
  ctx.save();
  ctx.globalAlpha = 0.5;
  ctx.lineWidth = 5;
  ctx.beginPath();
  _borderElement(ctx, stage.rootElement);
  ctx.stroke();
  ctx.restore();
}

Code void borderHitTest(Stage stage, Coordinate point) #

static void borderHitTest(Stage stage, Coordinate point){
  var ctx = stage.ctx;

  var hits = RetainedUtil.hitTest(stage, point);

  if(hits.length > 0){
    ctx.save();
    ctx.lineWidth = 2;

    hits.forEach((e) {
      _borderElement(ctx, e, true);
    });
    ctx.restore();
  }
}
