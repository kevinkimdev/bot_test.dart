Code bool draw() #

bool draw(){
  if (_ctx == null) {
    _ctx = _canvas.getContext('2d');
  } else {
    this._ctx.clearRect(0, 0, this._canvas.width, this._canvas.height);
  }

  return this._element.draw(this._ctx);
}
