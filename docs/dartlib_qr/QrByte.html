factory QrByte(String input) {
  requireArgumentNotNull(input, 'input');
  final charCodes = input.charCodes();
  for(final v in charCodes) {
    requireArgument(v < 255, 'ascii-only');
  }
  return new QrByte._internal(charCodes);
}
