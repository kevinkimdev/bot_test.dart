String toString() {
  StringBuffer sb = new StringBuffer();
  sb.add("DirectoryIOException");
  if (!message.isEmpty()) {
    sb.add(": $message");
    if (path != null) {
      sb.add(", path = $path");
    }
    if (osError != null) {
      sb.add(" ($osError)");
    }
  } else if (osError != null) {
    sb.add(": $osError");
    if (path != null) {
      sb.add(", path = $path");
    }
  }
  return sb.toString();
}
