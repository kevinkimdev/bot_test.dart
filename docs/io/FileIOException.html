String toString() {
  StringBuffer sb = new StringBuffer();
  sb.add("FileIOException");
  if (!message.isEmpty()) {
    sb.add(": $message");
    if (osError != null) {
      sb.add(" ($osError)");
    }
  } else if (osError != null) {
    sb.add(": osError");
  }
  return sb.toString();
}
