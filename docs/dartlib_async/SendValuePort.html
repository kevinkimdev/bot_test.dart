SendValuePort(this._func) {
  port.receive((TInput value, SendPort reply) {
    try {
      final TOutput output = _func(value);
      reply.send(new FutureValueResult<TOutput>(output));
    } catch(final ex) {
      reply.send(new FutureValueResult<TOutput>.exception(ex));
    }
  });
}
