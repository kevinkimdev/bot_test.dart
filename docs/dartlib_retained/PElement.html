}

Code AffineTransform getTransformToRoot() #

core.AffineTransform getTransformToRoot(){
  var tx = new core.AffineTransform();
  if(_parent != null){
    tx.concatenate(_parent.getTransformToRoot());
  }
  tx.concatenate(getTransform());
  return tx;
}
